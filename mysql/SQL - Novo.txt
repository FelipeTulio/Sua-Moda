drop database if exists suamoda;
create database suamoda;
use suamoda;

CREATE TABLE Administrador (
CPF varchar(11) PRIMARY KEY,
Nome varchar(100),
Senha varchar(100),
Email varchar(100)
)engine=innodb;

CREATE TABLE Produto (
Nome varchar(100),
Numero_Serie int PRIMARY KEY,
Descricao varchar(100),
Categoria varchar(100),
Preco float
)engine=innodb;

CREATE TABLE imagens (
idImagem int PRIMARY KEY,
idProduto int,
imagens varchar(100),
FOREIGN KEY(idProduto) REFERENCES Produto (Numero_Serie)
)engine=innodb;

CREATE TABLE cadastrar (
CPF varchar(11),
Numero_Serie int,
PRIMARY KEY(CPF,Numero_Serie),
FOREIGN KEY(CPF) REFERENCES Administrador (CPF),
FOREIGN KEY(Numero_Serie) REFERENCES Produto (Numero_Serie)
)engine=innodb;

CREATE TABLE Usuario (
CPF varchar(11) PRIMARY KEY,
Nome varchar(100),
Senha varchar(100),
Email varchar(100),
Sobrenome varchar(100),
rua varchar(100),
bairro varchar(100),
numero varchar(100),
Telefone varchar(100),
Cidade varchar(100),
Foto varchar(100)
)engine=innodb;

CREATE TABLE compra (
Numero_Serie int,
CPF varchar(11),
data varchar(100),
Numero_Venda int,
PRIMARY KEY(Numero_Serie,CPF),
FOREIGN KEY(Numero_Serie) REFERENCES Produto (Numero_Serie),
FOREIGN KEY(CPF) REFERENCES Usuario (CPF)
)engine=innodb;
